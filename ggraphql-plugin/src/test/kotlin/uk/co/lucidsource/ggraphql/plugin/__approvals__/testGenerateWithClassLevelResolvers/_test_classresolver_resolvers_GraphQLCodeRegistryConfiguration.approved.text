package test.classresolver.resolvers

import graphql.schema.FieldCoordinates
import graphql.schema.GraphQLCodeRegistry
import java.util.concurrent.Executor
import test.classresolver.datafetchers.MutationCreateUserDataFetcher
import test.classresolver.datafetchers.MutationDeleteUserDataFetcher
import test.classresolver.datafetchers.MutationUpdateUserDataFetcher
import test.classresolver.datafetchers.ProductOperationsCreateDataFetcher
import test.classresolver.datafetchers.ProductOperationsFindByIdDataFetcher
import test.classresolver.datafetchers.ProductOperationsFindByNameDataFetcher
import test.classresolver.datafetchers.ProductOperationsUpdateDataFetcher
import test.classresolver.datafetchers.QueryProductByIdDataFetcher
import test.classresolver.datafetchers.QueryProductsDataFetcher
import test.classresolver.datafetchers.QueryUsersDataFetcher
import uk.co.lucidsource.ggraphql.api.serde.Deserializer

public class GraphQLCodeRegistryConfiguration(
  public val productResolver: ProductResolver,
  public val queryResolver: QueryResolver,
  public val customProductResolver: CustomProductResolver,
  public val mutationResolver: MutationResolver,
  public val deserializer: Deserializer,
  public val executor: Executor,
) {
  public fun applyConfiguration(graphQLCodeRegistry: GraphQLCodeRegistry.Builder):
      GraphQLCodeRegistry.Builder {
    graphQLCodeRegistry.dataFetcher(FieldCoordinates.coordinates("ProductOperations", "findById"),
        ProductOperationsFindByIdDataFetcher(productResolver, deserializer, executor))
    graphQLCodeRegistry.dataFetcher(FieldCoordinates.coordinates("ProductOperations", "findByName"),
        ProductOperationsFindByNameDataFetcher(productResolver, deserializer, executor))
    graphQLCodeRegistry.dataFetcher(FieldCoordinates.coordinates("ProductOperations", "create"),
        ProductOperationsCreateDataFetcher(productResolver, deserializer, executor))
    graphQLCodeRegistry.dataFetcher(FieldCoordinates.coordinates("ProductOperations", "update"),
        ProductOperationsUpdateDataFetcher(productResolver, deserializer, executor))
    graphQLCodeRegistry.dataFetcher(FieldCoordinates.coordinates("Query", "users"),
        QueryUsersDataFetcher(queryResolver, deserializer, executor))
    graphQLCodeRegistry.dataFetcher(FieldCoordinates.coordinates("Query", "products"),
        QueryProductsDataFetcher(queryResolver, deserializer, executor))
    graphQLCodeRegistry.dataFetcher(FieldCoordinates.coordinates("Query", "productById"),
        QueryProductByIdDataFetcher(customProductResolver, deserializer, executor))
    graphQLCodeRegistry.dataFetcher(FieldCoordinates.coordinates("Mutation", "createUser"),
        MutationCreateUserDataFetcher(mutationResolver, deserializer, executor))
    graphQLCodeRegistry.dataFetcher(FieldCoordinates.coordinates("Mutation", "updateUser"),
        MutationUpdateUserDataFetcher(mutationResolver, deserializer, executor))
    graphQLCodeRegistry.dataFetcher(FieldCoordinates.coordinates("Mutation", "deleteUser"),
        MutationDeleteUserDataFetcher(mutationResolver, deserializer, executor))
    return graphQLCodeRegistry
  }
}
