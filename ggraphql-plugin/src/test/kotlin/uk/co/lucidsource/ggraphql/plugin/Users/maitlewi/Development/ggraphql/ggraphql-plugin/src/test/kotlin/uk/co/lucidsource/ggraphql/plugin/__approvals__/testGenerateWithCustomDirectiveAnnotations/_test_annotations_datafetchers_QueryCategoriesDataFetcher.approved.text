package test.annotations.datafetchers

import graphql.schema.DataFetcher
import graphql.schema.DataFetchingEnvironment
import java.util.concurrent.CompletableFuture
import java.util.concurrent.Executor
import kotlin.collections.List
import test.annotations.models.Category
import test.annotations.resolvers.CategoryResolver
import uk.co.lucidsource.ggraphql.api.serde.Deserializer

public class QueryCategoriesDataFetcher(
  public val service: CategoryResolver,
  public val deserializer: Deserializer,
  public val executor: Executor,
) : DataFetcher<CompletableFuture<List<Category>>> {
  override fun `get`(env: DataFetchingEnvironment): CompletableFuture<List<Category>> =
      CompletableFuture.supplyAsync({ service.categories(context = env.graphQlContext) }, executor)
}
