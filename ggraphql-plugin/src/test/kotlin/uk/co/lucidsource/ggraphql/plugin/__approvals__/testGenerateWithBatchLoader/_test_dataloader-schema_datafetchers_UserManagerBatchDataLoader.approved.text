package test.`dataloader-schema`.datafetchers

import graphql.GraphQLContext
import java.util.concurrent.CompletableFuture
import java.util.concurrent.CompletionStage
import java.util.concurrent.Executor
import kotlin.collections.List
import org.dataloader.BatchLoaderEnvironment
import org.dataloader.BatchLoaderWithContext
import test.`dataloader-schema`.models.User
import test.`dataloader-schema`.resolvers.UserResolver

public class UserManagerBatchDataLoader(
  public val service: UserResolver,
  public val executor: Executor,
) : BatchLoaderWithContext<User, User> {
  override fun load(keys: List<User>, batchLoaderEnvironment: BatchLoaderEnvironment):
      CompletionStage<List<User>> {

                    val context = batchLoaderEnvironment.getContext<GraphQLContext>()
                    return CompletableFuture.supplyAsync({ service.batchManager(keys, context =
        context) }, executor)
                
  }
}
