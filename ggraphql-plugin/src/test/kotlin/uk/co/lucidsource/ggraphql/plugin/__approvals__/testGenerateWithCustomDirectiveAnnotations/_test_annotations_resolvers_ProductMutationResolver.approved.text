package test.annotations.resolvers

import kotlin.Boolean
import kotlin.String
import org.springframework.transaction.`annotation`.Transactional
import test.annotations.models.CreateProductInput
import test.annotations.models.Product
import test.annotations.models.UpdateProductInput

public interface ProductMutationResolver {
  @Transactional
  public fun createProduct(input: CreateProductInput): Product?

  @Transactional
  public fun updateProduct(id: String, input: UpdateProductInput): Product?

  @Transactional
  public fun deleteProduct(id: String): Boolean?
}
