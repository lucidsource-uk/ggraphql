package test.classresolver.resolvers

import graphql.GraphQLContext
import kotlin.Float
import kotlin.String
import kotlin.collections.List
import test.classresolver.models.Product

public interface ProductResolver {
  public fun findById(id: String, context: GraphQLContext): Product?

  public fun findByName(name: String, context: GraphQLContext): List<Product>

  public fun create(
    name: String,
    price: Float,
    context: GraphQLContext,
  ): Product?

  public fun update(
    id: String,
    name: String?,
    price: Float?,
    context: GraphQLContext,
  ): Product?
}
