package test.`dataloader-schema`.datafetchers

import graphql.schema.DataFetcher
import graphql.schema.DataFetchingEnvironment
import java.util.concurrent.CompletableFuture
import java.util.concurrent.Executor
import test.`dataloader-schema`.models.User
import test.`dataloader-schema`.resolvers.UserResolver
import uk.co.lucidsource.ggraphql.api.serde.Deserializer

public class UserManagerDataFetcher(
  public val service: UserResolver,
  public val deserializer: Deserializer,
  public val executor: Executor,
) : DataFetcher<CompletableFuture<User>> {
  override fun `get`(env: DataFetchingEnvironment): CompletableFuture<User> {
    val dataLoader = env.getDataLoader<User, User>("UserManagerBatchDataLoader") 
    return dataLoader!!.load(env.getSource<User>()) 
  }
}
