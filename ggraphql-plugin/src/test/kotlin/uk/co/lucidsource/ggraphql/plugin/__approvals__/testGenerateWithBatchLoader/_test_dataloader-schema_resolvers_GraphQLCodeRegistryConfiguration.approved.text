package test.`dataloader-schema`.resolvers

import graphql.schema.FieldCoordinates
import graphql.schema.GraphQLCodeRegistry
import java.util.concurrent.Executor
import test.`dataloader-schema`.datafetchers.QueryUsersDataFetcher
import test.`dataloader-schema`.datafetchers.UserManagerDataFetcher
import uk.co.lucidsource.ggraphql.api.serde.Deserializer

public class GraphQLCodeRegistryConfiguration(
  public val userResolver: UserResolver,
  public val queryResolver: QueryResolver,
  public val deserializer: Deserializer,
  public val executor: Executor,
) {
  public fun applyConfiguration(graphQLCodeRegistry: GraphQLCodeRegistry.Builder):
      GraphQLCodeRegistry.Builder {
    graphQLCodeRegistry.dataFetcher(FieldCoordinates.coordinates("User", "manager"),
        UserManagerDataFetcher(userResolver, deserializer, executor))
    graphQLCodeRegistry.dataFetcher(FieldCoordinates.coordinates("Query", "users"),
        QueryUsersDataFetcher(queryResolver, deserializer, executor))
    return graphQLCodeRegistry
  }
}
